44 void GroupFileInfo::FromJson(const json& data)
45 {
46  this->name = Utils::GetValue(data, "name", "");
47  this->id = Utils::GetValue(data, "id", "");
48  this->path = Utils::GetValue(data, "path", "");
49  this->contact = Utils::GetValue(data, "contact", Group{});
50  if (Utils::HasValue(data, "parent"))
51  {
52  this->parent = std::make_unique<GroupFileInfo>();
53  this->parent->FromJson(data.at("parent"));
54  }
55  else
56  this->parent = nullptr;
57  this->isFile = Utils::GetValue(data, "isFile", false);
58  if (this->isFile)
59  {
60  this->file = FileInfo{};
61  this->file->FromJson(data);
62  }
63  else
64  this->file = std::nullopt;
65 
66  if (Utils::HasValue(data, "downloadInfo"))
67  this->DownloadUrl = Utils::GetOptional<std::string>(data.at("downloadInfo"), "url");
68  else
69  this->DownloadUrl = std::nullopt;
70 }
