27/**
28 * @brief QQ移动端头像戳一戳动作的对象
29 *
30 * Member Variable | Default Value
31 * --------------- | -------------
32 * `NudgeTarget::_kind` | `TargetKind::UNKNOWN`
33 * `NudgeTarget::_target` | `0_qq`
34 * `NudgeTarget::_subject` | `0`
35 */
36class NudgeTarget
37{
38public:
39 /**
40 * @brief 戳一戳类型
41 *
42 * `UNKNOWN` 为保留字段，使用时出现说明数据不合法
43 */
44 enum TargetKind : std::size_t
45 {
46 FRIEND = 0,
47 GROUP,
48 STRANGER,
49 UNKNOWN
50 };
51
52protected:
53 TargetKind _kind = TargetKind::UNKNOWN;
54 QQ_t _target;
55 int64_t _subject = 0;
56
57 static constexpr std::array<std::string_view, static_cast<std::size_t>(TargetKind::UNKNOWN)> _TargetKindStr = {
58 "Friend", "Group", "Stranger"};
59
60 static constexpr std::string_view _to_string(const TargetKind& m)
61 {
62 auto i = static_cast<std::size_t>(m);
63 if (i < _TargetKindStr.size()) return _TargetKindStr[i];
64 else
65 return "";
66 }
67
68 static constexpr TargetKind _to_enum(std::string_view s)
69 {
70 for (std::size_t i = 0; i < _TargetKindStr.size(); i++)
71 if (_TargetKindStr[i] == s) return static_cast<TargetKind>(i);
72
73 return TargetKind::UNKNOWN;
74 }
75
76public:
77 /**
78 * @brief 构造函数
79 *
80 */
81 NudgeTarget() : _kind(TargetKind::UNKNOWN), _target(0), _subject(0) {}
82
83 /**
84 * @brief 构造函数
85 *
86 * @param kind 戳一戳的类型
87 * @param target 戳一戳的对象
88 * @param subject 戳一戳的发送主体，群聊消息时为群号，私聊时为对方QQ号码
89 */
90 NudgeTarget(TargetKind kind, QQ_t target, UID_t subject) : _kind(kind), _target(target), _subject((int64_t)subject)
91 {
92 }
93
94 /**
95 * @brief 返回戳一戳类型
96 *
97 * @return 戳一戳类型 `enum`
98 */
99 TargetKind GetTargetKind() const { return this->_kind; }
100
101 /**
102 * @brief 返回戳一戳类型字符串
103 *
104 * @return 戳一戳类型 `std::string`
105 */
106 std::string GetTargetKindStr() const { return std::string(_to_string(this->_kind)); }
107
108 /**
109 * @brief 返回戳一戳的对象
110 *
111 * @return 戳一戳对象的QQ号
112 */
113 QQ_t GetTarget() const { return this->_target; }
114
115 /**
116 * @brief 返回戳一戳所在的群聊
117 *
118 * 仅在类型为 `TargetKind::GROUP` 时有效，否则返回 `0`
119 * @return 戳一戳的群号
120 */
121 GID_t GetGroup() const { return (this->_kind == TargetKind::GROUP) ? (GID_t)this->_subject : (GID_t)0; }
122
123 /**
124 * @brief 设置戳一戳类型
125 *
126 * @param kind 戳一戳类型 `enum`
127 * @return Reference to *this
128 */
129 NudgeTarget& SetTargetKind(TargetKind kind)
130 {
131 this->_kind = kind;
132 return *this;
133 }
134
135 /**
136 * @brief 设置戳一戳类型
137 *
138 * @param kind 戳一戳类型 `std::string`
139 * @return Reference to *this
140 */
141 NudgeTarget& SetTargetKind(const std::string& kind)
142 {
143 this->_kind = _to_enum(kind);
144 return *this;
145 }
146
147 /**
148 * @brief 设置戳一戳对象
149 *
150 * @param target 戳一戳对象QQ
151 * @return Reference to *this
152 */
153 NudgeTarget& SetTarget(QQ_t target)
154 {
155 this->_target = target;
156 return *this;
157 }
158
159 /**
160 * @brief 设置戳一戳的发送主体
161 *
162 * 不推荐，考虑使用 `NudgeFriend()`, `NudgeGroupMember()` 与 `NudgeStranger()`
163 * @param subject 戳一戳的发送主体
164 * @return Reference to *this
165 */
166 NudgeTarget& SetSubject(int64_t subject)
167 {
168 this->_subject = subject;
169 return *this;
170 }
171
172
173 /**
174 * @brief 设置为好友戳一戳消息
175 *
176 * @param target 好友QQ
177 * @return Reference to *this
178 */
179 NudgeTarget& NudgeFriend(QQ_t target)
180 {
181 this->_kind = TargetKind::FRIEND;
182 this->_target = target;
183 return *this;
184 }
185
186 /**
187 * @brief 设置为群聊戳一戳消息
188 *
189 * @param target 群友QQ
190 * @param group 群聊号码
191 * @return Reference to *this
192 */
193 NudgeTarget& NudgeGroupMember(QQ_t target, GID_t group)
194 {
195 this->_kind = TargetKind::GROUP;
196 this->_target = target;
197 this->_subject = (int64_t)group;
198 return *this;
199 }
200
201 /**
202 * @brief 设置为单向好友戳一戳消息
203 *
204 * @param target 陌生人QQ
205 * @return Reference to *this
206 */
207 NudgeTarget& NudgeStranger(QQ_t target)
208 {
209 this->_kind = TargetKind::STRANGER;
210 this->_target = target;
211 return *this;
212 }
213};
