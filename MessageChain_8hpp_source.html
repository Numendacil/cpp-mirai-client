63  template<typename T> class _has_type_

65  typedef char yes_type;

112  *

189  /**
190  * @brief 获取指定下标出的消息
191  *
192  * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性。
193  * 若消息类型与实际类型不匹配会抛出 `TypeDismatch` 异常
194  * @tparam MessageType 消息类型
195  * @param i 下标
196  * @return 该消息元素的引用
197  */
198  template<typename MessageType> MessageType& GetAt(size_type i) const
199  {
200  _type_check_<MessageType>();
201 
202  const auto& m = this->_message.at(i);
203  if (!m) throw std::runtime_error("Unexpected null pointer in MessageChain::at");
204  if (m->GetType() != MessageType::_TYPE_)
205  throw TypeDismatch(std::string(m->GetType()), std::string(MessageType::_TYPE_));
206  return *static_cast<MessageType*>(m.get());
207  }
208 
209  /**
210  * @brief 获取指定下标出的消息
211  *
212  * 约等于 `MessageChain::GetAt<MessageBase>(i)`
213  * @param i 下标
214  * @return 该消息元素的引用，类型为 `MessageBase&`
215  */
216  MessageBase& GetAt(size_type i) const
217  {
218  const auto& m = this->_message.at(i);
219  if (!m) throw std::runtime_error("Unexpected null pointer in MessageChain::at");
220  return *m;
221  }

223  /**
224  * @brief 获取所有指定类型的消息元素
225  *
226  * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性。
227  * @tparam MessageType 消息类型
228  * @return `std::vector<MessageType>`
229  */
230  template<typename MessageType> std::vector<MessageType> GetAll() const
231  {
232  _type_check_<MessageType>();
233 
234  std::vector<MessageType> v;
235  for (const auto& p : this->_message)
236  {
237  if (p->GetType() == MessageType::_TYPE_) v.emplace_back(*static_cast<MessageType*>(p.get()));
238  }
239  return v;
240  }
