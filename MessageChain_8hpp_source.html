53 * @brief 消息链对象，由一系列消息元素组成

63 template<typename T> class _has_type_
64 {
65 using yes_type = char;
66 using no_type = long;
67 template<typename U> static yes_type test(decltype(&U::_TYPE_));
68 template<typename U> static no_type test(...);
69
70 public:
71 static constexpr bool value = sizeof(test<T>(0)) == sizeof(yes_type);
72 };

75 {
76 static_assert(std::is_base_of<MessageBase, Message>::value, "Message must be a derived class of MessageBase");

78 };

110 /**
111 * @brief 在消息链结尾添加消息元素
112 *
113 * 类似 `std::vector::push_back()`。
114 * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性
115 * @tparam MessageType 消息类型
116 * @param m 要添加的消息
117 * @return reference to *this
118 */
119 template<typename MessageType> MessageChain& Append(MessageType&& m)
120 {
121 _type_check_<MessageType>();
122
123 this->_message.push_back(std::make_unique<MessageType>(std::forward<MessageType>(m)));
124 return *this;
125 }
126
127 /**
128 * @brief 在消息链结尾直接构造消息元素
129 *
130 * 类似 `std::vector::emplace_back()`。
131 * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性
132 * @tparam MessageType 消息类型
133 * @tparam Args 参数类型
134 * @param args 构造消息元素的参数
135 * @return reference to *this
136 */
137 template<typename MessageType, typename... Args> MessageChain& Append(Args&&... args)
138 {
139 _type_check_<MessageType>();
140
141 this->_message.push_back(std::make_unique<MessageType>(std::forward<Args>(args)...));
142 return *this;
143 }

189 /**
190 * @brief 获取指定下标出的消息
191 *
192 * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性。
193 * 若消息类型与实际类型不匹配会抛出 `TypeDismatch` 异常
194 * @tparam MessageType 消息类型
195 * @param i 下标
196 * @return 该消息元素的引用
197 */
198 template<typename MessageType> MessageType& GetAt(size_type i) const
199 {
200 _type_check_<MessageType>();
201
202 const auto& m = this->_message.at(i);
203 if (!m) throw std::runtime_error("Unexpected null pointer in MessageChain::at");
204 if (m->GetType() != MessageType::_TYPE_)
205 throw TypeDismatch(std::string(m->GetType()), std::string(MessageType::_TYPE_));
206 return *static_cast<MessageType*>(m.get());
207 }
208
209 /**
210 * @brief 获取指定下标出的消息
211 *
212 * 约等于 `MessageChain::GetAt<MessageBase>(i)`
213 * @param i 下标
214 * @return 该消息元素的引用，类型为 `MessageBase&`
215 */
216 MessageBase& GetAt(size_type i) const
217 {
218 const auto& m = this->_message.at(i);
219 if (!m) throw std::runtime_error("Unexpected null pointer in MessageChain::at");
220 return *m;
221 }
222
223 /**
224 * @brief 获取所有指定类型的消息元素
225 *
226 * 模版参数 `MessageType` 必须为 `MessageBase` 的派生类，且定义了 `MessageType::_TYPE_` 属性。
227 * @tparam MessageType 消息类型
228 * @return `std::vector<MessageType>`
229 */
230 template<typename MessageType> std::vector<MessageType> GetAll() const
231 {
232 _type_check_<MessageType>();
233
234 std::vector<MessageType> v;
235 for (const auto& p : this->_message)
236 {
237 if (p->GetType() == MessageType::_TYPE_) v.emplace_back(*static_cast<MessageType*>(p.get()));
238 }
239 return v;
240 }
