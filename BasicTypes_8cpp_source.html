45constexpr std::array<std::string_view, static_cast<std::size_t>(SEX::UNKNOWN) + 1> sex_names = {"MALE", "FEMALE",
46 "UNKNOWN"};
47
48constexpr std::string_view sex_to_str(const SEX& m)
49{
50 auto i = static_cast<std::size_t>(m);
51 if (i < sex_names.size()) return sex_names.at(i);
52 else
53 return "UNKNOWN";
54}
55
56constexpr SEX str_to_sex(std::string_view s)
57{
58 for (std::size_t i = 0; i < sex_names.size(); i++)
59 if (sex_names.at(i) == s) return static_cast<SEX>(i);
60
61 return SEX::UNKNOWN;
62}
63
64} // namespace

66void to_json(nlohmann::json& j, const SEX& p)
67{
68 j = sex_to_str(p);
69}
70void from_json(const nlohmann::json& j, SEX& p)
71{
72 p = str_to_sex(j.get<std::string>());
73}

78static constexpr std::array<std::string_view, static_cast<std::size_t>(PERMISSION::UNKNOWN)> permission_names = {
79 "OWNER", "ADMINISTRATOR", "MEMBER"};
80
81constexpr std::string_view permission_to_str(const PERMISSION& m)
82{
83 auto i = static_cast<std::size_t>(m);
84 if (i < permission_names.size()) return permission_names.at(i);
85 else
86 return "";
87}
88
89constexpr PERMISSION str_to_permission(std::string_view s)
90{
91 for (std::size_t i = 0; i < permission_names.size(); i++)
92 if (permission_names.at(i) == s) return static_cast<PERMISSION>(i);
93
94 return PERMISSION::UNKNOWN;
95}
