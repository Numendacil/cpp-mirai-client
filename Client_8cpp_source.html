63{
64 if (this != &rhs)
65 {
66 std::lock_guard<std::mutex> lk(this->_mtx);
67 this->_config = std::move(rhs._config);
68 this->_SessionKey = std::move(rhs._SessionKey);
69 this->_SessionKeySet = rhs._SessionKeySet.load();
70 this->_connected = rhs._connected;
71 this->_HttpClient = std::move(rhs._HttpClient);
72 this->_MessageClient = std::move(rhs._MessageClient);
73 this->_ThreadPool = std::move(rhs._ThreadPool);
74 }
75 return *this;
76}

80void MiraiClient::Connect()
81{
82 std::unique_lock<std::mutex> lk(this->_mtx);
83 if (this->_connected) return;
84
85 LOG_DEBUG(this->_GetLogger(), "Connecting to Mirai-Api-Http, initializing clients and threadpool");
86
87 this->_SessionKey = "";
88 this->_SessionKeySet = false;
89
90 this->_ThreadPool = std::make_unique<Utils::ThreadPool>(this->_config.ThreadPoolSize);
91
92 LOG_DEBUG(this->_GetLogger(), "Threadpool initialized");
93
94 this->_HttpClient = std::make_unique<Details::HttpClientImpl>(
95 this->_config.HttpUrl, this->_config.ConnectionTimeout, this->_config.ReadTimeout, this->_config.WriteTimeout);
96
97 this->_MessageClient = std::make_unique<Details::MessageClientImpl>(
98 this->_config.HeartbeatInterval, this->_config.HandshakeTimeout, this->_config.EnablePong,
99 this->_config.EnableDeflate, this->_config.AutoReconnect, this->_config.MinRetryInterval,
100 this->_config.MaxRetryInterval);
101
102 this->_MessageClient->OnOpen(
103 [this](const ix::WebSocketOpenInfo& info)
104 {
105 LOG_DEBUG(
106 this->_GetLogger(),
107 "MessageClient connected with response header: " + [&]() -> std::string
108 {
109 std::string header;
110 for (const auto& p : info.headers)
111 header += "\n" + p.first + ": " + p.second;
112 return header;
113 }());
114
115 this->_HandshakeInfo = {info.uri, {info.headers.begin(), info.headers.end()}, info.protocol};
116 });
117
118 this->_MessageClient->OnError(
119 [this](const ix::WebSocketErrorInfo& info)
120 {
121 LOG_DEBUG(
122 this->_GetLogger(),
123 "MessageClient failed to connect with error: " + [&]() -> std::string
124 {
125 std::string msg;
126 msg += "\nHTTP Status: " + std::to_string(info.http_status);
127 msg += "\nReason: " + info.reason;
128 msg += "\nDecompressionError: ";
129 msg += info.decompressionError ? "True" : "False";
130 return msg;
131 }());
132
133 auto callback = this->_GetErrorCallback();
134 if (callback)
135 {
136 ClientConnectionErrorEvent event{info.retries, info.wait_time, info.http_status, info.reason,
137 info.decompressionError};
138 callback(event);
139 }
140 });
141
142 this->_MessageClient->OnClose(
143 [this](const ix::WebSocketCloseInfo& info)
144 {
145 LOG_DEBUG(
146 this->_GetLogger(),
147 "MessageClient connection closed: " + [&]() -> std::string
148 {
149 std::string msg;
150 msg += "\nCode: " + std::to_string(info.code);
151 msg += "\nReason: " + info.reason;
152 msg += "\nClosedByRemote: ";
153 msg += info.remote ? "True" : "False";
154 return msg;
155 }());
156
157 {
158 std::lock_guard<std::mutex> lk(this->_mtx);
159 this->_connected = false;
160 }
161 this->_SessionKeySet = false;
162 auto callback = this->_GetClosedCallback();
163 if (callback)
164 {
165 ClientConnectionClosedEvent event{info.code, info.reason, info.remote};
166 callback(event);
167 }
168 });
169
170 this->_MessageClient->OnText(
171 [this](const std::string& message)
172 {
173 LOG_TRACE(this->_GetLogger(), "Received: " + message);
174
175 try
176 {
177 json msg = json::parse(message);
178
179 if (!msg.contains("data")) return;
180
181 /*std::string id = msg.value("syncId", "");*/
182 json data = msg.at("data");
183 if (!data.is_object()) return;
184
185 json resp = Utils::ParseResponse(data);
186
187 if (!this->_SessionKeySet) // check for sessionKey
188 {
189 if (this->_ReadSessionKey(data))
190 {
191 return;
192 }
193 }
194
195 this->_DispatchEvent(data);
196 }
197 catch (const ParseError& e)
198 {
199 auto callback = this->_GetParseErrorCallback();
200 if (this->_ParseErrorCallback) this->_ParseErrorCallback({e, message});
201 }
202 catch (const std::exception& e)
203 {
204 auto callback = this->_GetParseErrorCallback();
205 if (this->_ParseErrorCallback) this->_ParseErrorCallback({ParseError{e.what(), message}, message});
206 }
207 catch (...)
208 {
209 auto callback = this->_GetParseErrorCallback();
210 if (this->_ParseErrorCallback)
211 this->_ParseErrorCallback({ParseError{"Unknown error", message}, message});
212 }
213 });
214
215 LOG_DEBUG(this->_GetLogger(), "Clients initialized");
216
217 LOG_DEBUG(this->_GetLogger(),
218 "Calling connect from MessageClient: " + this->_config.WebsocketUrl
219 + "/all?verifyKey=" + this->_config.VerifyKey + "&qq=" + this->_config.BotQQ.to_string());
220
221 this->_MessageClient->Connect(this->_config.WebsocketUrl + "/all?verifyKey=" + this->_config.VerifyKey
222 + "&qq=" + this->_config.BotQQ.to_string());
223
224 if (!this->_cv.wait_for(lk, std::chrono::seconds(10), [this]() -> bool { return this->_connected; }))
225 {
226 LOG_DEBUG(this->_GetLogger(), "Timeout waiting for session key, closing down");
227
228 // failed to get sessionKey
229 this->_MessageClient->Disconnect();
230 throw NetworkException(-2, "Failed to receive session key from server");
231 }
232
233 LOG_DEBUG(this->_GetLogger(), "Successfully connected");
234}

236void MiraiClient::Disconnect()
237{
238 std::unique_lock<std::mutex> lk(this->_mtx);
239 if (this->_MessageClient)
240 {
241 this->_MessageClient->Disconnect();
242 LOG_DEBUG(this->_GetLogger(), "Successfully disconnected");
243 }
244 this->_ThreadPool = nullptr;
245
246 LOG_DEBUG(this->_GetLogger(), "Threadpool shutdown complete");
247
248 this->_HttpClient = nullptr;
249 this->_MessageClient = nullptr;
250
251 LOG_DEBUG(this->_GetLogger(), "Clients shutdown complete");
252}

254bool MiraiClient::_ReadSessionKey(const json& data)
255{
256 if (!(data.contains("session") && data.at("session").is_string())) return false;
257
258 std::string session = data.at("session").get<std::string>();
259
260 LOG_DEBUG(this->_GetLogger(), "Session key obtained: " + session);
261
262 // Check http connection
263 LOG_DEBUG(this->_GetLogger(), "Checking HTTP connection, calling /sessionInfo with session key: " + session);
264
265 json resp = this->_HttpClient->SessionInfo(session);
266
267 LOG_DEBUG(this->_GetLogger(), "Checking HTTP connection, received: " + resp.dump());
268
269 std::string validate = Utils::GetValue(resp, "sessionKey", "");
270 if (session != validate)
271 throw MiraiApiHttpException(-1, "Dismatched sessionKey: \"" + session + "\" <-> \"" + validate);
272
273 ClientConnectionEstablishedEvent event;
274 {
275 std::unique_lock<std::mutex> lk(this->_mtx);
276 this->_SessionKey = session;
277 this->_SessionKeySet = true;
278 this->_connected = true;
279 this->_HandshakeInfo.SessionKey = session;
280 this->_HandshakeInfo.BotProfile = Utils::GetValue(resp, "qq", User{});
281 event = this->_HandshakeInfo;
282 }
283
284 auto callback = this->_GetEstablishedCallback();
285 if (callback) callback(event);
286
287 this->_cv.notify_one();
288 return true;
289}

291void MiraiClient::_DispatchEvent(const json& data)
292{
293 if (!(data.contains("type") && data.at("type").is_string())) return;
294
295 std::string type = data.at("type").get<std::string>();
296
297 LOG_DEBUG(this->_GetLogger(), "Dispatching message, type: " + type);
298
299 EventHandler handler;
300 {
301 std::lock_guard<std::mutex> lk(this->_mtx);
302 if (this->_EventHandlers.count(type)) handler = this->_EventHandlers.at(type);
303 }
304 if (handler)
305 {
306 LOG_DEBUG(this->_GetLogger(), "Found handler");
307 (void)this->_ThreadPool->enqueue(
308 [data, handler, this]()
309 {
310 try
311 {
312 handler(data);
313 }
314 catch (const ParseError& e)
315 {
316 auto callback = this->_GetParseErrorCallback();
317 if (this->_ParseErrorCallback) this->_ParseErrorCallback({e, data.dump()});
318 }
319 });
320 return;
321 }
322 LOG_DEBUG(this->_GetLogger(), "No matching handler found");
323}

678GroupFileInfo MiraiClient::UploadGroupFile(GID_t GroupId, const string& UploadPath, const string& name,
679 std::istream& file)
680{
681 string s;
682 char buffer[4096];
683 while (file.read(buffer, sizeof(buffer)))
684 s.append(buffer, sizeof(buffer));
685 s.append(buffer, file.gcount());
686 return this->UploadGroupFile(GroupId, UploadPath, name, s);
687}
