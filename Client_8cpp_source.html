81 void MiraiClient::Connect()
82 {
83  if (this->_connected) return;
84 
85  this->_SessionKey = "";
86  this->_SessionKeySet = false;
87  this->_HttpClient = std::make_unique<Details::HttpClientImpl>(
88  this->_config.HttpUrl, this->_config.ConnectionTimeout, this->_config.ReadTimeout, this->_config.WriteTimeout);
89 
90  this->_MessageClient = std::make_unique<Details::MessageClientImpl>(
91  this->_config.HeartbeatInterval, this->_config.HandshakeTimeout, this->_config.EnablePong,
92  this->_config.EnableDeflate, this->_config.AutoReconnect, this->_config.MinRetryInterval,
93  this->_config.MaxRetryInterval);
94 
95  this->_ThreadPool = std::make_unique<Utils::ThreadPool>(this->_config.ThreadPoolSize);
96 
97  this->_MessageClient->OnOpen(
98  [this](const ix::WebSocketOpenInfo& info)
99  {
100  this->_connected = true;
101  this->_HandshakeInfo = {info.uri, {info.headers.begin(), info.headers.end()}, info.protocol};
102  });
103 
104  this->_MessageClient->OnError(
105  [this](const ix::WebSocketErrorInfo& info)
106  {
107  auto callback = this->_GetErrorCallback();
108  if (callback)
109  {
110  ClientConnectionErrorEvent event{info.retries, info.wait_time, info.http_status, info.reason,
111  info.decompressionError};
112  callback(event);
113  }
114  });
115 
116  this->_MessageClient->OnClose(
117  [this](const ix::WebSocketCloseInfo& info)
118  {
119  this->_connected = false;
120  this->_SessionKeySet = false;
121  auto callback = this->_GetClosedCallback();
122  if (callback)
123  {
124  ClientConnectionClosedEvent event{info.code, info.reason, info.remote};
125  callback(event);
126  }
127  });
128 
129  this->_MessageClient->OnText(
130  [this](const std::string& message)
131  {
132  try
133  {
134  json msg = json::parse(message);
135 
136  if (!msg.contains("data")) return;
137 
138  /*std::string id = msg.value("syncId", "");*/
139  json data = msg.at("data");
140  if (!data.is_object()) return;
141 
142  (void)Utils::ParseResponse(data);
143 
144  if (!this->_SessionKeySet) // check for sessionKey
145  {
146  if (this->_ReadSessionKey(data))
147  {
148  return;
149  }
150  }
151 
152  this->_DispatchEvent(data);
153  }
154  catch (const ParseError& e)
155  {
156  auto callback = this->_GetParseErrorCallback();
157  if (this->_ParseErrorCallback) this->_ParseErrorCallback({e, message});
158  }
159  catch (const std::exception& e)
160  {
161  auto callback = this->_GetParseErrorCallback();
162  if (this->_ParseErrorCallback) this->_ParseErrorCallback({ParseError{e.what(), message}, message});
163  }
164  catch (...)
165  {
166  auto callback = this->_GetParseErrorCallback();
167  if (this->_ParseErrorCallback)
168  this->_ParseErrorCallback({ParseError{"Unknown error", message}, message});
169  }
170  });
171 
172  this->_MessageClient->Connect(this->_config.WebsocketUrl + "/all?verifyKey=" + this->_config.VerifyKey
173  + "&qq=" + this->_config.BotQQ.to_string());
174 
175  {
176  std::unique_lock<std::mutex> lk(this->_mtx);
177  if (!this->_cv.wait_for(lk, std::chrono::seconds(10), [this]() -> bool { return this->_SessionKeySet; }))
178  {
179  // failed to get sessionKey
180  this->_MessageClient->Disconnect();
181  throw NetworkException(-2, "Failed to receive session key from server");
182  }
183  }
184 }

194  this->_MessageClient = nullptr;

197 bool MiraiClient::_ReadSessionKey(const json& data)
198 {
199  if (!(data.contains("session") && data.at("session").is_string())) return false;
200 
201  std::string session = data.at("session").get<std::string>();
202 
203  // Check http connection
204  json resp = this->_HttpClient->SessionInfo(session);
205  std::string validate = Utils::GetValue(resp, "sessionKey", "");
206  if (session != validate)
207  throw MiraiApiHttpException(-1, "Dismatched sessionKey: \"" + session + "\" <-> \"" + validate);
208  this->_HandshakeInfo.SessionKey = session;
209  this->_HandshakeInfo.BotProfile = Utils::GetValue(resp, "qq", User{});
210 
211  {
212  std::unique_lock<std::mutex> lk(this->_mtx);
213  this->_SessionKey = session;
214  this->_SessionKeySet = true;
215  }
216 
217  auto callback = this->_GetEstablishedCallback();
218  if (callback) callback(this->_HandshakeInfo);
219 
220  this->_cv.notify_one();
221  return true;
222 }

224 void MiraiClient::_DispatchEvent(const json& data)
225 {
226  if (!(data.contains("type") && data.at("type").is_string())) return;
227 
228  std::string type = data.at("type").get<std::string>();
229  EventHandler handler;
230  {
231  std::lock_guard<std::mutex> lk(this->_mtx);
232  if (this->_EventHandlers.count(type)) handler = this->_EventHandlers.at(type);
233  }
234  if (handler)
235  {
236  (void)this->_ThreadPool->enqueue(
237  [data, handler, this]()
238  {
239  try
240  {
241  handler(data);
242  }
243  catch (const ParseError& e)
244  {
245  auto callback = this->_GetParseErrorCallback();
246  if (this->_ParseErrorCallback) this->_ParseErrorCallback({e, data.dump()});
247  }
248  catch (const std::exception& e)
249  {
250  auto callback = this->_GetParseErrorCallback();
251  if (this->_ParseErrorCallback)
252  this->_ParseErrorCallback({ParseError{e.what(), data.dump()}, data.dump()});
253  }
254  catch (...)
255  {
256  auto callback = this->_GetParseErrorCallback();
257  if (this->_ParseErrorCallback)
258  this->_ParseErrorCallback({ParseError{"Unknown error", data.dump()}, data.dump()});
259  }
260  });
261  }
