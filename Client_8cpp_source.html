118void MiraiClient::Connect()
119{
120 if (!this->adaptor_) throw std::runtime_error("Null pointer: No adaptor is set");
121
122 if (this->connected_) return;
123
124 LOG_DEBUG(*(this->logger_), "Connecting to Mirai-Api-Http, initializing threadpool");
125
126 this->pool_ = std::make_unique<Utils::ThreadPool>(this->PoolSize_);
127
128 LOG_DEBUG(*(this->logger_), "Threadpool initialized");
129
130 this->adaptor_->OnConnectionEstablished(
131 [this](ClientConnectionEstablishedEvent event)
132 {
133 LOG_DEBUG(
134 *(this->logger_),
135 "MessageClient connected with response header: " + [&]() -> std::string
136 {
137 std::string header;
138 for (const auto& p : event.headers)
139 header += "\n" + p.first + ": " + p.second;
140 return header;
141 }());
142 {
143 std::unique_lock<std::shared_mutex> lk(this->mtx_);
144 this->connected_ = true;
145 this->SessionKey_ = event.SessionKey;
146 }
147 if (this->ConnectionEstablishedCallback_) this->ConnectionEstablishedCallback_(std::move(event));
148 });
149
150 this->adaptor_->OnConnectionClosed(
151 [this](ClientConnectionClosedEvent event)
152 {
153 LOG_DEBUG(
154 *(this->logger_),
155 "MessageClient connection closed: " + [&]() -> std::string
156 {
157 std::string msg;
158 msg += "\nCode: " + std::to_string(event.code);
159 msg += "\nReason: " + event.reason;
160 msg += "\nClosedByRemote: ";
161 msg += event.remote ? "True" : "False";
162 return msg;
163 }());
164 {
165 std::unique_lock<std::shared_mutex> lk(this->mtx_);
166 this->connected_ = false;
167 }
168 if (this->ConnectionClosedCallback_) this->ConnectionClosedCallback_(std::move(event));
169 });
170
171 this->adaptor_->OnConnectionError(
172 [this](ClientConnectionErrorEvent event)
173 {
174 LOG_DEBUG(
175 *(this->logger_),
176 "MessageClient failed to connect with error: " + [&event]() -> std::string
177 {
178 std::string msg;
179 msg += "\nHTTP Status: " + std::to_string(event.HttpStatus);
180 msg += "\nReason: " + event.reason;
181 msg += "\nDecompressionError: ";
182 msg += event.DecompressionError ? "True" : "False";
183 return msg;
184 }());
185 if (this->ConnectionErrorCallback_) this->ConnectionErrorCallback_(std::move(event));
186 });
187
188 this->adaptor_->OnRecv(
189 [this](const std::string& message)
190 {
191 LOG_TRACE(*(this->logger_), "Received: " + message);
192
193 try
194 {
195 json msg = json::parse(message);
196 if (!msg.contains("data")) return;
197
198 /*std::string id = msg.value("syncId", "");*/
199 json data = msg.at("data");
200 if (!data.is_object()) return;
201 data = Utils::ParseResponse(data);
202 if (!(data.contains("type") && data.at("type").is_string())) return;
203
204 auto type = data.at("type").get<EventTypes>();
205 LOG_TRACE(*(this->logger_), "Dispatching message, type: " + to_string(type));
206
207 EventHandler handler;
208 {
209 if (this->EventHandlers_.count(type)) handler = this->EventHandlers_.at(type);
210 }
211 std::visit(
212 [&data, this](auto&& handler)
213 {
214 using FuncType = std::decay_t<decltype(handler)>;
215 using EventType = typename traits::function_traits<FuncType>::type;
216
217 if (!handler)
218 {
219 LOG_TRACE(*(this->logger_), "No matching handler found");
220 return;
221 }
222
223 LOG_TRACE(*(this->logger_), "Found handler");
224
225 if constexpr (!std::disjunction_v< std::is_same<EventType, ClientConnectionEstablishedEvent>,
226 std::is_same<EventType, ClientConnectionErrorEvent>,
227 std::is_same<EventType, ClientConnectionClosedEvent>,
228 std::is_same<EventType, ClientParseErrorEvent> >)
229 {
230 EventType event;
231 event.SetClient_(this);
232 from_json(std::move(data), event);
233 (void)this->pool_->enqueue(handler, std::move(event));
234 }
235 },
236 handler);
237 }
238 catch (const ParseError& e)
239 {
240 if (this->ParseErrorCallback_) this->ParseErrorCallback_({{}, e, message});
241 }
242 catch (const std::exception& e)
243 {
244 if (this->ParseErrorCallback_) this->ParseErrorCallback_({{}, ParseError{e.what(), message}, message});
245 }
246 catch (...)
247 {
248 if (this->ParseErrorCallback_)
249 this->ParseErrorCallback_({{}, ParseError{"Unknown error", message}, message});
250 }
251 });
252
253 LOG_DEBUG(*(this->logger_), "Calling connect");
254
255 (void)this->adaptor_->Connect();
256
257 LOG_DEBUG(*(this->logger_), "Successfully connected");
258}

260void MiraiClient::Disconnect()
261{
262 this->adaptor_->Disconnect(this->GetSessionKey_());
263 LOG_DEBUG(*(this->logger_), "Successfully disconnected");
264
265 this->pool_ = nullptr;
266 LOG_DEBUG(*(this->logger_), "Threadpool shutdown complete");
267}

288FriendMessageEvent MiraiClient::GetFriendMessage(MessageId_t id, QQ_t qq) const
289{
290 auto msg = this->adaptor_->MessageFromId(this->GetSessionKey_(), id, qq);
291
292 if (!std::holds_alternative<FriendMessageEvent>(msg))
293 throw TypeDismatchError(to_string(FriendMessageEvent::GetType()),
294 to_string(std::visit([](auto&& m) -> EventTypes { return m.GetType(); }, msg)));
295
296 return std::get<FriendMessageEvent>(msg);
297}

496GroupFileInfo MiraiClient::UploadGroupFile(GID_t GroupId, string UploadPath, string name,
497 std::istream& file) const
498{
499 string s;
500 char buffer[4096]; // NOLINT(*-avoid-c-arrays)
501 while (file.read(buffer, sizeof(buffer))) // NOLINT(*-array-to-pointer-decay)
502 s.append(buffer, sizeof(buffer)); // NOLINT(*-array-to-pointer-decay)
503 s.append(buffer, file.gcount()); // NOLINT(*-array-to-pointer-decay)
504 return this->UploadGroupFile(GroupId, std::move(UploadPath), std::move(name), std::move(s));
505}
