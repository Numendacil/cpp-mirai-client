35MessageChain::MessageChain(const MessageChain& rhs)
36{
37 MessageContainer v;
38 v.reserve(rhs._message.size());

114std::optional<MessageChain::SourceInfo> MessageChain::GetSourceInfo() const
115{
116 for (const auto& p : this->_message)
117 {
118 if (p->GetType() == SourceMessage::_TYPE_)
119 {
120 auto s = static_cast<SourceMessage*>(p.get());
121 return SourceInfo{s->GetMessageId(), s->GetTimestamp()};
122 }
123 }
124 return std::nullopt;
125}

170void MessageChain::RemoveInvalid()
171{
172 this->_message.erase(
173 std::remove_if(this->_message.begin(), this->_message.end(), [](const value_type& p) { return !p->isValid(); }),
174 this->_message.end());
175}

177bool MessageChain::isValid() const
178{
179 bool empty = true;
180 for (const auto& p : this->_message)
181 {
182 if (!p->isValid()) return false;
183 if (empty && !p->ToJson().empty()) empty = false;
184 }
185 return !empty;
186}

188void MessageChain::FromJson(const json& data)
189{
190 if (!data.is_array()) return;
191 this->_message.clear();
192 this->_message.reserve(data.size());
193 for (const auto& p : data)
194 {
195 if (!p.is_object() || !p.contains("type") || !p["type"].is_string()) continue;
196 std::string type = p["type"].get<std::string>();
197 if (!MessageFactory.Exist(type)) continue;
198 this->_message.emplace_back(MessageFactory.Make(type))->FromJson(p);
199 }
200}
201
202json MessageChain::ToJson(bool ignoreInvalid) const
203{
204 json data = json::array();
205 for (const auto& p : this->_message)
206 {
207 if (ignoreInvalid && !p->isValid()) continue;
208 json msg = p->ToJson();
209 if (!msg.empty() && msg.contains("type")) data += msg;
210 }
211 return data;
212}
