188 void MessageChain::FromJson(const json& data)
189 {
190  if (!data.is_array()) return;
191  this->_message.clear();
192  this->_message.reserve(data.size());
193  for (const auto& p : data)
194  {
195  if (!p.is_object() || !p.contains("type") || !p["type"].is_string()) continue;
196  std::string type = p["type"].get<std::string>();
197  if (!MessageFactory.Exist(type)) continue;
198  this->_message.emplace_back(MessageFactory.Make(type))->FromJson(p);
199  }
200 }

202 json MessageChain::ToJson(bool ignoreInvalid) const
203 {
204  json data = json::array();
205  for (const auto& p : this->_message)
206  {
207  if (ignoreInvalid && !p->isValid()) continue;
208  json msg = p->ToJson();
209  if (!msg.empty() && msg.contains("type")) data += msg;
210  }
211  return data;
212 }
