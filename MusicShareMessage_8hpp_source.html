65 static constexpr std::array<std::string_view, static_cast<std::size_t>(MusicKind::UNKNOWN)> _MusicKindStr = {

68 static constexpr std::string_view _to_string(const MusicKind& m)
69 {
70 auto i = static_cast<std::size_t>(m);
71 if (i < _MusicKindStr.size()) return _MusicKindStr[i];
72 else
73 return "";
74 }
75
76 static constexpr MusicKind _to_enum(std::string_view s)
77 {
78 for (std::size_t i = 0; i < _MusicKindStr.size(); i++)
79 if (_MusicKindStr[i] == s) return static_cast<MusicKind>(i);
80
81 return MusicKind::UNKNOWN;
82 }
