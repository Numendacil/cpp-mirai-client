29 /**
30  * @brief 连接建立事件
31  *
32  * 会在连接成功建立后， `MiraiClient::Connect()` 返回之前广播。
33  * `MiraiClient` 的mirai相关api应在此事件之后才能被正常使用。
34  */
35 struct ClientConnectionEstablishedEvent
36 {
37  /// 连接地址
38  std::string uri;
39  /// 接收的文件头信息
40  std::map<std::string, std::string> headers;
41  /// 协议信息
42  std::string protocol;
43  /// 本次连接的session key
44  std::string SessionKey;
45  /// 获取到的的Bot资料
46  User BotProfile;
47 };

49 /**
50  * @brief 连接错误事件
51  *
52  * 通常出现于网络问题导致的无法连接目标服务器。
53  * 连接建立后出现的错误将导致 `ClientConnectionClosedEvent`
54  */
55 struct ClientConnectionErrorEvent
56 {
57  /// 当前重试次数
58  uint32_t RetryCount = 0;
59  /// 下一次重试前的等待时间
60  double WaitTime = 0;
61  /// Http连接状态
62  int HttpStatus = 0;
63  /// 错误原因
64  std::string reason;
65  /// 是否为解压缩编码错误
66  bool DecompressionError = false;
67 };
68 
69 /**
70  * @brief 连接丢失事件
71  *
72  * 只可能在 `ClientConnectionEstablishedEvent` 之后发生，表明已建立的连接因为某种原因关闭了。
73  * 若是 `MiraiClient::Disconnect()` 导致的连接关闭，该事件将会在函数返回之前被广播。
74  * 连接关闭后不再应该进行任何mirai api的使用，直到下次连接建立接收到 `ClientConnectionEstablishedEvent` 为止。
75  * 若开启了自动重连选项，`MiraiClient` 会在非客户端主动关闭导致的连接丢失后尝试自动重连。重连过程中的错误将会触发 `ClientConnectionErrorEvent`
76  */
77 struct ClientConnectionClosedEvent
78 {
79  /// 信息码
80  uint16_t code;
81  /// 连接关闭原因
82  std::string reason;
83  /// 是否由远程服务器关闭的连接
84  bool remote;
85 };

91  */
92 struct ClientParseErrorEvent

95  ParseError error;
