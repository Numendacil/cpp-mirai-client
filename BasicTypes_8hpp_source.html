40/**
41 * @brief Base class for QQ_t and GID_t
42 *
43 * 不允许实例化，仅作为基类提供基本功能
44 */
45class UID_t
46{
47protected:
48 int64_t _number;
49
50 UID_t() : _number(0) {}
51 explicit UID_t(int64_t num) : _number(num) {}
52
53public:
54 explicit operator int64_t() const { return this->_number; }
55 std::string to_string() const { return std::to_string(this->_number); }
56
57 bool operator<(const UID_t& uid) const { return this->_number < uid._number; }
58 bool operator==(const UID_t& uid) const { return this->_number == uid._number; }
59
60 bool operator!=(const UID_t& uid) const { return !(*this == uid); }
61 bool operator>(const UID_t& uid) const { return uid < *this; }
62 bool operator>=(const UID_t& uid) const { return !(*this < uid); }
63 bool operator<=(const UID_t& uid) const { return !(uid < *this); }
64
65 friend void to_json(nlohmann::json&, const UID_t&);
66 friend void from_json(const nlohmann::json&, UID_t&);
67};
