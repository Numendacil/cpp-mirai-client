106 /**
107  * @brief 文件路径
108  *
109  * 类似于Union的结构，可由文件id或者文件位置 `/path/to/file` 指定。
110  * 由于文件名可以不唯一，推荐尽量使用id。
111  *
112  * Member Variable | Default Value
113  * --------------- | -------------
114  * `FilePath::_dir` | `""`
115  * `FilePath::_isPath` | `false`
116  */
117 class FilePath
118 {
119 protected:
120  std::string _dir;
121  bool _isPath = false;
122 
123 public:
124  /**
125  * @brief 构造函数
126  *
127  */
128  FilePath() : _isPath(false) {}
129 
130  /**
131  * @brief 构造函数
132  *
133  * @param dir 文件路径，可能为id或文件名路径
134  * @param isPath 是否为文件名路径
135  */
136  FilePath(const std::string& dir, bool isPath = false) : _dir(dir), _isPath(isPath) {}
137 
138  /**
139  * @brief 构造函数
140  * @param file `GroupFileInfo` 对象
141  */
142  FilePath(const GroupFileInfo& file) : _dir(file.id), _isPath(false) {}
143 
144  /**
145  * @brief 判定是否为文件名路径
146  *
147  * @return `bool`
148  */
149  bool isPath() const { return this->_isPath; }
150 
151  /**
152  * @brief 获取文件名路径
153  *
154  * 仅当 `isPath()` 为 `true` 时有效，否则返回空字符串
155  * @return 文件名路径
156  */
157  std::string GetPath() const { return this->_isPath ? this->_dir : ""; }
158 
159  /**
160  * @brief 获取文件id
161  *
162  * 仅当 `isPath()` 为 `false` 时有效，否则返回空字符串
163  * @return 文件id
164  */
165  std::string GetId() const { return this->_isPath ? "" : this->_dir; }
166 
167  /**
168  * @brief 设置文件名路径
169  *
170  * @param path 文件名路径
171  * @return Reference to *this
172  */
173  FilePath& SetPath(const std::string& path)
174  {
175  this->_isPath = true;
176  this->_dir = path;
177  return *this;
178  }
179 
180  /**
181  * @brief 设置文件id
182  *
183  * @param id 文件名id
184  * @return Reference to *this
185  */
186  FilePath& SetId(const std::string& id)
187  {
188  this->_isPath = false;
189  this->_dir = id;
190  return *this;
191  }

194 /**

197  * 发送时可以指定图片id，图片链接，图片路径或图片的base64编码，

199  * Member Variable | Default Value

203  * `MiraiImage::path` | `""`
204  * `MiraiImage::base64` | `""`
205  * `MiraiImage::width` | `0`
206  * `MiraiImage::height` | `0`
207  * `MiraiImage::size` | `0`
208  * `MiraiImage::ImageType` | `""`
209  * `MiraiImage::isEmoji` | `false`
210  */
211 struct MiraiImage : public Serializable
212 {
213  /// 图片id，从mirai获得
214  std::string id;
215  /// 图片链接
216  std::string url;
217  /// 图片路径
218  std::string path;
219  /// 图片base64编码
220  std::string base64;
221  /// 图片宽度
222  int width = 0;
223  /// 图片长度
224  int height = 0;
225  /// 图片大小
226  int64_t size = 0;
227  /// 图片类型（JPEG，PNG，……）
228  std::string
229  ImageType; // TODO: replace with enum, see https://github.com/mamoe/mirai/blob/dev/mirai-core-api/src/commonMain/kotlin/message/data/Image.kt#L493
230  /// 是否为表情
231  bool isEmoji = false;
232 
233  MiraiImage(const std::string& id = {}, const std::string& url = {}, const std::string& path = {},
