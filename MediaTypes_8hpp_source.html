107/**
108 * @brief 文件路径
109 *
110 * 类似于Union的结构，可由文件id或者文件位置 `/path/to/file` 指定。
111 * 由于文件名可以不唯一，推荐尽量使用id。
112 *
113 * Member Variable | Default Value
114 * --------------- | -------------
115 * `FilePath::_dir` | `""`
116 * `FilePath::_isPath` | `false`
117 */
118class FilePath
119{
120protected:
121 std::string _dir;
122 bool _isPath = false;
123
124public:
125 /**
126 * @brief 构造函数
127 *
128 */
129 FilePath() = default;
130
131 /**
132 * @brief 构造函数
133 *
134 * @param dir 文件路径，可能为id或文件名路径
135 * @param isPath 是否为文件名路径
136 */
137 FilePath(std::string dir, bool isPath = false) : _dir(std::move(dir)), _isPath(isPath) {}
138
139 /**
140 * @brief 构造函数
141 * @param file `GroupFileInfo` 对象
142 */
143 FilePath(const GroupFileInfo& file) : _dir(file.id) {}
144
145 /**
146 * @brief 判定是否为文件名路径
147 *
148 * @return `bool`
149 */
150 bool isPath() const { return this->_isPath; }
151
152 /**
153 * @brief 获取文件名路径
154 *
155 * 仅当 `isPath()` 为 `true` 时有效，否则返回空字符串
156 * @return 文件名路径
157 */
158 std::string GetPath() const { return this->_isPath ? this->_dir : ""; }
159
160 /**
161 * @brief 获取文件id
162 *
163 * 仅当 `isPath()` 为 `false` 时有效，否则返回空字符串
164 * @return 文件id
165 */
166 std::string GetId() const { return this->_isPath ? "" : this->_dir; }
167
168 /**
169 * @brief 设置文件名路径
170 *
171 * @param path 文件名路径
172 * @return Reference to *this
173 */
174 FilePath& SetPath(const std::string& path)
175 {
176 this->_isPath = true;
177 this->_dir = path;
178 return *this;
179 }
180
181 /**
182 * @brief 设置文件id
183 *
184 * @param id 文件名id
185 * @return Reference to *this
186 */
187 FilePath& SetId(const std::string& id)
188 {
189 this->_isPath = false;
190 this->_dir = id;
191 return *this;
192 }
193};

195/**
196 * @brief Mirai图片
197 *
198 * 发送时可以指定图片id，图片链接，图片路径或图片的base64编码，
199 * 优先级为 id > url > path > base64，所以设置某一项属性前最好确保前面的域为空。
200 * Member Variable | Default Value
201 * --------------- | -------------
202 * `MiraiImage::id` | `""`
203 * `MiraiImage::url` | `""`
204 * `MiraiImage::path` | `""`
205 * `MiraiImage::base64` | `""`
206 * `MiraiImage::width` | `0`
207 * `MiraiImage::height` | `0`
208 * `MiraiImage::size` | `0`
209 * `MiraiImage::ImageType` | `""`
210 * `MiraiImage::isEmoji` | `false`
211 */
212struct MiraiImage : public Serializable
213{
214 /// 图片id，从mirai获得
215 std::string id;
216 /// 图片链接
217 std::string url;
218 /// 图片路径
219 std::string path;
220 /// 图片base64编码
221 std::string base64;
222 /// 图片宽度
223 int width = 0;
224 /// 图片长度
225 int height = 0;
226 /// 图片大小
227 int64_t size = 0;
228 /// 图片类型（JPEG，PNG，……）
229 std::string
230 ImageType; // TODO: replace with enum, see https://github.com/mamoe/mirai/blob/dev/mirai-core-api/src/commonMain/kotlin/message/data/Image.kt#L493
231 /// 是否为表情
232 bool isEmoji = false;
233
234 MiraiImage(std::string id = {}, std::string url = {}, std::string path = {}, std::string base64 = {})
235 : id(std::move(id)), url(std::move(url)), path(std::move(path)), base64(std::move(base64))
236 {
237 }
238
239 /**
240 * @brief 检查对象能否用于发送
241 *
242 * @return `bool`
243 */
244 bool isValid() const
245 {
246 return !(this->id.empty() && this->url.empty() && this->path.empty() && this->base64.empty());
247 }
248
249 void FromJson(const nlohmann::json&) override;
250 nlohmann::json ToJson() const override;
251};

254using GroupImage = MiraiImage;
