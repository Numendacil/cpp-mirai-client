66  * @brief 提供与mirai-api-http网络交互的封装
67  *
68  * 所有与mirai的交互都需要通过该类来实现。使用时可以通过调用http

70  *

109  template<typename T> class _has_type_
110  {
111  typedef char yes_type;
112  typedef long no_type;
113  template<typename U> static yes_type test(decltype(&U::_TYPE_));
114  template<typename U> static no_type test(...);
115 
116  public:
117  static constexpr bool value = sizeof(test<T>(0)) == sizeof(yes_type);
118  };

120  template<typename Event> constexpr static void _type_check_()
121  {
122  static_assert(std::is_base_of<EventBase, Event>::value, "Event must be a derived class of MessageBase");
123  static_assert(_has_type_<Event>::value, "Event must contain a static atrribute _TYPE_");
124  };

177  /**
178  * @brief 注册事件回调函数
179  *
180  * 模版参数 `EventType` 必须为 `EventBase` 的派生类，且定义了 `EventType::_TYPE_` 属性
181  * @tparam EventType 事件类型
182  * @param callback 回调函数
183  */
184  template<typename EventType> void On(EventCallback<EventType> callback)
185  {
186  _type_check_<EventType>();
187  std::lock_guard<std::mutex> lk(this->_mtx);
188  this->_EventHandlers[std::string(EventType::_TYPE_)] = [callback, this](const nlohmann::json& j)
189  {
190  EventType event(this);
191  event.FromJson(j);
192  callback(event);
193  };
194  }
