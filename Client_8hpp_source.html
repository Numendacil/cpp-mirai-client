65/**
66 * @brief 提供与mirai-api-http网络交互的封装
67 *
68 * 所有与mirai的交互都需要通过该类来实现。使用时可以通过调用http
69 * api以及注册事件回调函数来完成消息的处理与发送。
70 *
71 * 连接设置保存在 `SessionConfigs` 类中，更改设置后需要重新连接mirai来使新配置生效。
72 *
73 * 所有非常量方法均有可能会抛出异常，异常的种类见 `Exceptions.hpp`。
74 */

111 template<typename T> class _has_type_
112 {
113 using yes_type = char;
114 using no_type = long;
115 template<typename U> static yes_type test(decltype(&U::_TYPE_));
116 template<typename U> static no_type test(...);
117
118 public:
119 static constexpr bool value = sizeof(test<T>(0)) == sizeof(yes_type);
120 };

122 template<typename Event> constexpr static void _type_check_()
123 {
124 static_assert(std::is_base_of<EventBase, Event>::value, "Event must be a derived class of MessageBase");
125 static_assert(_has_type_<Event>::value, "Event must contain a static atrribute _TYPE_");
126 };

187 /**
188 * @brief 注册事件回调函数
189 *
190 * 模版参数 `EventType` 必须为 `EventBase` 的派生类，且定义了 `EventType::_TYPE_` 属性
191 * @tparam EventType 事件类型
192 * @param callback 回调函数
193 */
194 template<typename EventType> void On(EventCallback<EventType> callback)
195 {
196 _type_check_<EventType>();
197 std::lock_guard<std::mutex> lk(this->_mtx);
198 this->_EventHandlers[std::string(EventType::_TYPE_)] = [callback, this](const nlohmann::json& j)
199 {
200 EventType event(this);
201 event.FromJson(j);
202 callback(event);
203 };
204 }
